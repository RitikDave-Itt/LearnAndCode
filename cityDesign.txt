function designCity() {
    planZones();
    buildTransport();
    createBuildings();
    setUpUtilities();
    addAmenities();
    ensureEnvironmentCare();
    growEconomy();
    ImplementLaws();
}

function planZones() {
    defineAreas();
    addGreenSpaces();
    layoutCity();
}

function buildTransport() {
    makeRoads();
    makeRailLines();
    setUpPublicTransport();
    addTransportHubs();
}

function createBuildings() {
    buildHouses();
    buildShopsAndOffices();
}

function setUpUtilities() {
    supplyWater();
    manageWaste();
    provideElectricity();
    setUpCommunicationInfra();
    
}

function addCommonFacilities() {
    buildSchools();
    setUpHospitals();
    makeParks();
    provideSafety();
}

function environmentCare() {
    protectNature();
    useGreenMethods();
    makeEcoFriendlyBuildings();
}

function growEconomy() {
    setIndustrialAreas();
    boostTech();
    buildTouristSpots();
}

function ImplementLaws() {
    setupPoliceStructure();
    setupArmy();
    setUpEmergencyTaskForce();
}
